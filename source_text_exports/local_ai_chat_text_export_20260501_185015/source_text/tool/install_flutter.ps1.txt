# tool/install_flutter.ps1
#
# One-time Flutter SDK installer for Windows/PowerShell agents.
# Idempotent. Installs into $env:FLUTTER_HOME or $HOME\flutter, then prepends
# bin to PATH for the current process.

$ErrorActionPreference = 'Stop'

$flutterHome = if ($env:FLUTTER_HOME) {
  $env:FLUTTER_HOME
} else {
  Join-Path $HOME 'flutter'
}

$flutterChannel = if ($env:FLUTTER_CHANNEL) {
  $env:FLUTTER_CHANNEL
} else {
  'stable'
}

$existing = Get-Command flutter -ErrorAction SilentlyContinue
if ($existing) {
  Write-Host "Flutter already on PATH at: $($existing.Source)"
  flutter --version
  exit 0
}

if (-not (Test-Path $flutterHome)) {
  Write-Host "==> Cloning Flutter ($flutterChannel) into $flutterHome"
  git clone --depth 1 -b $flutterChannel https://github.com/flutter/flutter.git $flutterHome
} else {
  Write-Host "==> Flutter dir exists at $flutterHome; skipping clone"
}

$env:Path = (Join-Path $flutterHome 'bin') + [IO.Path]::PathSeparator + $env:Path

# First flutter invocation triggers Dart SDK download. Fail loudly here rather
# than letting a half-installed Flutter mask the real error later.
Write-Host "==> Bootstrapping Flutter (downloads Dart SDK on first run; ~5 min)"
flutter --version
if ($LASTEXITCODE -ne 0) {
  Write-Error @"
Flutter bootstrap failed (exit $LASTEXITCODE).
Common causes:
  - No internet / firewall blocks storage.googleapis.com
  - Behind a corporate proxy → set HTTP_PROXY / HTTPS_PROXY first
  - In a region where Google Storage is restricted → set:
      `$env:FLUTTER_STORAGE_BASE_URL = 'https://storage.flutter-io.cn'
      `$env:PUB_HOSTED_URL          = 'https://pub.flutter-io.cn'
"@
  exit $LASTEXITCODE
}

Write-Host "==> Disabling analytics"
flutter config --no-analytics --no-cli-animations | Out-Null

Write-Host ""
Write-Host "Flutter installed at: $flutterHome"
Write-Host "For this PowerShell session:"
Write-Host "`$env:Path = `"$flutterHome\bin;`$env:Path`""

# tool/install_tesseract.ps1
#
# Idempotent Tesseract installer for Windows. Installs Tesseract OCR via
# winget (UB Mannheim build), then verifies that the chosen languages have
# trained data installed. Adds Tesseract to PATH for the current process.
#
# Env vars:
#   OCR_LANGS   plus-separated language codes to verify
#               (default: chi_tra+eng; common: chi_tra+chi_sim+eng+jpn)
#
# After this runs, `dart run bin/index.dart books\some.cbz` should work.

$ErrorActionPreference = 'Stop'

$langSpec = if ($env:OCR_LANGS) { $env:OCR_LANGS } else { 'chi_tra+eng' }
$wantedLangs = $langSpec -split '\+' | Where-Object { $_ -ne '' }

function Add-TesseractToPath {
  $candidates = @(
    'C:\Program Files\Tesseract-OCR',
    "$env:LOCALAPPDATA\Programs\Tesseract-OCR"
  )
  foreach ($p in $candidates) {
    if ((Test-Path $p) -and ($env:Path -notlike "*$p*")) {
      $env:Path = $env:Path + [IO.Path]::PathSeparator + $p
    }
  }
}

function Get-TessdataDir {
  $cmd = Get-Command tesseract -ErrorAction SilentlyContinue
  if (-not $cmd) { return $null }
  $base = Split-Path -Parent $cmd.Source
  $direct = Join-Path $base 'tessdata'
  if (Test-Path $direct) { return $direct }
  # UB Mannheim default: <install>\tessdata
  return $direct  # caller can mkdir it
}

# --- Main ---

Add-TesseractToPath

if (-not (Get-Command tesseract -ErrorAction SilentlyContinue)) {
  Write-Host '==> Installing Tesseract (UB Mannheim, ~50 MB)' -ForegroundColor Cyan
  if (Get-Command winget -ErrorAction SilentlyContinue) {
    & winget install --id=UB-Mannheim.TesseractOCR --silent `
      --accept-source-agreements --accept-package-agreements
    if ($LASTEXITCODE -ne 0) {
      throw "winget install failed (exit $LASTEXITCODE)"
    }
  } else {
    Write-Error @'
winget not available. Install Tesseract manually:
  https://github.com/UB-Mannheim/tesseract/wiki
Or via scoop:  scoop install tesseract
'@
    exit 1
  }
  Add-TesseractToPath
}

if (-not (Get-Command tesseract -ErrorAction SilentlyContinue)) {
  Write-Error 'Tesseract installed but not on PATH. Restart shell or check install dir.'
  exit 1
}

Write-Host '==> tesseract --version'
& tesseract --version

# Verify trained data files for requested languages.
$tessdata = Get-TessdataDir
$installedRaw = (& tesseract --list-langs) 2>&1
$installedLangs = $installedRaw |
  Where-Object { $_ -notmatch '^List of available languages' -and $_.Trim() -ne '' } |
  ForEach-Object { $_.Trim() }

$missing = @()
foreach ($l in $wantedLangs) {
  if ($installedLangs -notcontains $l) { $missing += $l }
}

if ($missing.Count -gt 0) {
  Write-Host ''
  Write-Host "==> Missing trained data: $($missing -join ', ')" -ForegroundColor Yellow
  Write-Host "Downloading from tesseract-ocr/tessdata into: $tessdata"
  if (-not (Test-Path $tessdata)) {
    New-Item -ItemType Directory -Force -Path $tessdata | Out-Null
  }
  foreach ($l in $missing) {
    $url = "https://github.com/tesseract-ocr/tessdata/raw/main/$l.traineddata"
    $dst = Join-Path $tessdata "$l.traineddata"
    Write-Host "   $l ..."
    Invoke-WebRequest -Uri $url -OutFile $dst
  }
  Write-Host ''
  Write-Host '==> Re-listing languages'
  & tesseract --list-langs
} else {
  Write-Host ''
  Write-Host "All requested languages installed: $($wantedLangs -join ', ')" -ForegroundColor Green
}

Write-Host ''
Write-Host 'Tesseract is ready.' -ForegroundColor Green
Write-Host ''
Write-Host 'Smoke test:'
Write-Host '  dart run bin/index.dart books\some_manga.cbz'
Write-Host '  (or any .png / .jpg page)'

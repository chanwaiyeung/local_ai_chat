import 'dart:convert';

import 'package:flutter_test/flutter_test.dart';
import 'package:http/http.dart' as http;
import 'package:http/testing.dart';
import 'package:local_ai_chat/services/embedding_service.dart';

void main() {
  group('EmbeddingService.embedAll', () {
    test('empty input returns empty without network call', () async {
      final service = EmbeddingService(
        baseUrl: 'http://invalid.local',
        httpClient: MockClient((_) async {
          fail('should not call network');
        }),
      );

      expect(await service.embedAll([]), isEmpty);
    });

    test('uses /api/embed batch endpoint when available', () async {
      var batchCalls = 0;
      var legacyCalls = 0;

      final mock = MockClient((request) async {
        if (request.url.path == '/api/embed') {
          batchCalls++;
          final body = jsonDecode(request.body) as Map<String, dynamic>;
          final inputs = (body['input'] as List).cast<String>();
          return http.Response(
            jsonEncode({
              'embeddings': [
                for (var i = 0; i < inputs.length; i++) [i.toDouble(), 0, 0],
              ],
            }),
            200,
            headers: {'content-type': 'application/json'},
          );
        }

        if (request.url.path == '/api/embeddings') {
          legacyCalls++;
        }
        return http.Response('not found', 404);
      });

      final service = EmbeddingService(httpClient: mock);
      final embeddings = await service.embedAll(['a', 'b', 'c']);

      expect(batchCalls, 1);
      expect(legacyCalls, 0);
      expect(embeddings, hasLength(3));
      expect(embeddings[0], [0.0, 0.0, 0.0]);
      expect(embeddings[2], [2.0, 0.0, 0.0]);
    });

    test('falls back to /api/embeddings when /api/embed returns 404', () async {
      var batchCalls = 0;
      var legacyCalls = 0;

      final mock = MockClient((request) async {
        if (request.url.path == '/api/embed') {
          batchCalls++;
          return http.Response('not implemented', 404);
        }

        if (request.url.path == '/api/embeddings') {
          legacyCalls++;
          final body = jsonDecode(request.body) as Map<String, dynamic>;
          final prompt = body['prompt'] as String;
          return http.Response(
            jsonEncode({
              'embedding': [prompt.length.toDouble(), 0, 0],
            }),
            200,
            headers: {'content-type': 'application/json'},
          );
        }

        return http.Response('unexpected', 500);
      });

      final service = EmbeddingService(httpClient: mock);
      final embeddings = await service.embedAll(['ab', 'cdef']);

      expect(batchCalls, 1);
      expect(legacyCalls, 2);
      expect(embeddings, hasLength(2));
      expect(embeddings[0], [2.0, 0.0, 0.0]);
      expect(embeddings[1], [4.0, 0.0, 0.0]);
    });

    test('falls back when /api/embed returns wrong shape', () async {
      var batchCalls = 0;
      var legacyCalls = 0;

      final mock = MockClient((request) async {
        if (request.url.path == '/api/embed') {
          batchCalls++;
          return http.Response(
            jsonEncode({
              'embeddings': [
                [1.0, 0, 0],
              ],
            }),
            200,
            headers: {'content-type': 'application/json'},
          );
        }

        if (request.url.path == '/api/embeddings') {
          legacyCalls++;
          return http.Response(
            jsonEncode({
              'embedding': [9.0, 0, 0],
            }),
            200,
            headers: {'content-type': 'application/json'},
          );
        }

        return http.Response('unexpected', 500);
      });

      final service = EmbeddingService(httpClient: mock);
      final embeddings = await service.embedAll(['a', 'b']);

      expect(batchCalls, 1);
      expect(legacyCalls, 2);
      expect(embeddings, [
        [9.0, 0.0, 0.0],
        [9.0, 0.0, 0.0],
      ]);
    });

    test('parallel legacy fallback respects concurrency cap', () async {
      var inFlight = 0;
      var maxObserved = 0;

      final mock = MockClient((request) async {
        if (request.url.path == '/api/embed') {
          return http.Response('not implemented', 404);
        }

        inFlight++;
        if (inFlight > maxObserved) {
          maxObserved = inFlight;
        }
        await Future<void>.delayed(const Duration(milliseconds: 10));
        inFlight--;

        return http.Response(
          jsonEncode({
            'embedding': [1.0],
          }),
          200,
          headers: {'content-type': 'application/json'},
        );
      });

      final service = EmbeddingService(httpClient: mock);
      await service.embedAll(List.filled(20, 'x'));

      expect(maxObserved, lessThanOrEqualTo(4));
      expect(maxObserved, greaterThanOrEqualTo(2));
    });

    test('throws when /api/embed returns server error', () async {
      final mock = MockClient((_) async {
        return http.Response('boom', 500);
      });
      final service = EmbeddingService(httpClient: mock);

      expect(
        () => service.embedAll(['a']),
        throwsA(isA<Exception>()),
      );
    });
  });
}

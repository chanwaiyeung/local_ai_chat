// lib/server/api_server.dart
//
// Local AI Server — Phase 1 of 智讀館 (AI Library).
//
// Changes vs. previous draft:
//  A. /query now actually calls the LLM via the injected `generate` callback.
//  B. /query/stream truly streams: each LLM token becomes one SSE delta event.
//  C. LLM is decoupled via the LlmStreamGenerator typedef
//     (see ollama_client.dart for a concrete impl).
//  D. Auth failure returns 401 Unauthorized (was 403 Forbidden).
//  E. Middleware order fixed: CORS now wraps the error handler, so error
//     responses carry CORS headers (browsers/webviews otherwise see a CORS
//     error instead of the JSON error body).
//  F. _localIps() lists ALL non-loopback IPv4 with interface names so you
//     can see which IP your phone should hit, instead of guessing one.
//  G. SSE response includes `x-accel-buffering: no` to defeat proxy buffering.
//  H. Open-server warning printed when authToken is null/empty.
//  I. Body size capped (default 8 KiB) to avoid memory blowup on huge POSTs.
//  J. Hits capped at topK (default 6) so the prompt can't blow LLM context.
//  K. /health returns JSON for parser symmetry on the mobile client.
//  L. Citations payload built explicitly (no hidden Chunk.toJson() coupling).

import 'dart:async';
import 'dart:convert';
import 'dart:io';
import 'dart:typed_data';

import 'package:shelf/shelf.dart';
import 'package:shelf/shelf_io.dart' as io;
import 'package:shelf_cors_headers/shelf_cors_headers.dart';
import 'package:shelf_router/shelf_router.dart';

import '../services/rag_service.dart';
import '../services/vector_store.dart';

/// Streaming LLM generator. Yields text deltas from a prompt. Throws on error.
typedef LlmStreamGenerator = Stream<String> Function(String prompt);

class ApiServer {
  static const _jsonHeaders = {
    'content-type': 'application/json; charset=utf-8',
  };
  static const _sseHeaders = {
    'content-type': 'text/event-stream; charset=utf-8',
    'cache-control': 'no-cache',
    'connection': 'keep-alive',
    'x-accel-buffering': 'no',
  };

  final RagService rag;
  final VectorStore store;
  final LlmStreamGenerator generate;
  final String? authToken;
  final int topK;
  final int maxBodyBytes;

  ApiServer({
    required this.rag,
    required this.store,
    required this.generate,
    this.authToken,
    this.topK = 6,
    this.maxBodyBytes = 8 * 1024,
  });

  // ----------------------------- Routing -----------------------------

  Router get _router {
    final router = Router();

    // Pre-flight (also handled by CORS middleware; this is belt-and-suspenders).
    router.options('/<ignored|.*>', (_) => Response.ok(''));

    // Health (no auth so health checks don't need secrets).
    router.get(
        '/health',
        (_) => _jsonResponse({
              'status': 'ok',
              'time': DateTime.now().toIso8601String(),
            }));

    // List indexed documents.
    router.get('/docs', _withAuth((Request req) async {
      return _jsonResponse({'docs': store.docNames});
    }));

    // Non-streaming query.
    router.post('/query', _withAuth((Request req) async {
      final data = await _readJsonObject(req);
      final query = _requiredString(data, 'query');
      final docName = _optionalString(data, 'docName');

      final hits =
          (await rag.retrieve(query, docName: docName)).take(topK).toList();
      final prompt = _buildPrompt(query, hits);

      final buf = StringBuffer();
      await for (final delta in generate(prompt)) {
        buf.write(delta);
      }

      return _jsonResponse({
        'answer': buf.toString(),
        'citations': _citations(hits),
      });
    }));

    // Streaming query (Server-Sent Events).
    router.post('/query/stream', _withAuth((Request req) async {
      final data = await _readJsonObject(req);
      final query = _requiredString(data, 'query');
      final docName = _optionalString(data, 'docName');

      final hits =
          (await rag.retrieve(query, docName: docName)).take(topK).toList();
      final prompt = _buildPrompt(query, hits);

      return Response(
        HttpStatus.ok,
        headers: _sseHeaders,
        body: _streamSse(prompt, hits),
      );
    }));

    // Read-mode: list every chunk of a single doc, in chunk-index order.
    // The mobile UI uses this to show source text without re-running RAG.
    // Path-param routes can't go through _withAuth (signature mismatch),
    // so we call _checkAuth manually.
    router.get('/docs/<docName|.*>/chunks',
        (Request req, String docName) async {
      final fail = _checkAuth(req);
      if (fail != null) return fail;
      final decoded = Uri.decodeComponent(docName);
      final chunks = store.chunksOf(decoded);
      return _jsonResponse({
        'docName': decoded,
        'chunks': chunks
            .map((c) => {
                  'docName': c.docName,
                  'chunkIndex': c.chunkIndex,
                  'text': c.text,
                })
            .toList(),
      });
    });

    // Pure retrieve (no LLM call). Useful for: (a) pre-flighting which
    // chunks would be used; (b) retrieve-first UIs where the mobile
    // client may render its own prompt later. Caller can override topK.
    router.post('/rag/retrieve', _withAuth((Request req) async {
      final data = await _readJsonObject(req);
      final query = _requiredString(data, 'query');
      final docName = _optionalString(data, 'docName');
      final requested = _optionalInt(data, 'topK') ?? defaultRetrieveTopK;
      // Clamp to the server's configured maximum so callers can't ask for
      // unbounded result sets.
      final effectiveTopK = requested.clamp(1, topK);

      final hits = (await rag.retrieve(query, docName: docName))
          .take(effectiveTopK)
          .toList();

      return _jsonResponse({
        'query': query,
        'docName': docName,
        'topK': effectiveTopK,
        'hits': _citations(hits),
      });
    }));

    return router;
  }

  /// Default top-K for `POST /rag/retrieve` when the caller doesn't
  /// specify one. Aligned with the server's main [topK] (6) so that
  /// retrieve-first clients have enough candidates for any client-side
  /// re-ranking they may want to do. Callers can override up to the
  /// server's configured [topK] (constructor argument).
  static const int defaultRetrieveTopK = 6;

  // ----------------------------- Auth -----------------------------

  Handler _withAuth(Handler handler) {
    return (Request req) async {
      final fail = _checkAuth(req);
      if (fail != null) return fail;
      return handler(req);
    };
  }

  /// Returns null if auth passes, or a 401 [Response] to short-circuit
  /// the handler. Used by routes with path parameters where shelf_router
  /// expects a multi-argument handler that [_withAuth] can't wrap.
  Response? _checkAuth(Request req) {
    // No token configured → server is intentionally open (a warning was
    // printed at start). Skip the check. `trim()` keeps a token of just
    // spaces from being treated as "configured".
    if (authToken == null || authToken!.trim().isEmpty) return null;
    final auth = req.headers['authorization'];
    if (auth == null || !auth.startsWith('Bearer ')) {
      return _unauthorized('Missing Bearer token');
    }
    if (auth.substring(7) != authToken) {
      return _unauthorized('Invalid token');
    }
    return null;
  }

  Response _unauthorized(String reason) =>
      _jsonResponse({'error': reason}, statusCode: 401);

  // ----------------------------- Body parsing -----------------------------

  Future<Map<String, dynamic>> _readJsonObject(Request req) async {
    final cl = req.contentLength;
    if (cl != null && cl > maxBodyBytes) {
      throw const FormatException('Request body too large.');
    }

    final builder = BytesBuilder(copy: false);
    await for (final chunk in req.read()) {
      builder.add(chunk);
      if (builder.length > maxBodyBytes) {
        throw const FormatException('Request body too large.');
      }
    }

    final body = utf8.decode(builder.toBytes());
    if (body.isEmpty) {
      throw const FormatException('Request body is empty.');
    }

    final dynamic decoded;
    try {
      decoded = jsonDecode(body);
    } on FormatException {
      throw const FormatException('Request body must be valid JSON.');
    }

    if (decoded is Map<String, dynamic>) return decoded;
    throw const FormatException('Request body must be a JSON object.');
  }

  String _requiredString(Map<String, dynamic> data, String key) {
    final v = data[key];
    if (v is String && v.trim().isNotEmpty) return v.trim();
    throw FormatException('"$key" must be a non-empty string.');
  }

  String? _optionalString(Map<String, dynamic> data, String key) {
    final v = data[key];
    if (v == null) return null;
    if (v is String && v.trim().isNotEmpty) return v.trim();
    throw FormatException('"$key" must be a string when provided.');
  }

  int? _optionalInt(Map<String, dynamic> data, String key) {
    final v = data[key];
    if (v == null) return null;
    if (v is int) return v;
    throw FormatException('"$key" must be an integer when provided.');
  }

  // ----------------------------- Streaming -----------------------------

  Stream<List<int>> _streamSse(String prompt, List<ScoredChunk> hits) async* {
    // Citations first so the UI can render them while tokens trickle in.
    yield _sseEvent({'type': 'citations', 'citations': _citations(hits)});
    await _yieldToSocket();
    try {
      await for (final delta in generate(prompt)) {
        yield _sseEvent({'type': 'delta', 'text': delta});
        await _yieldToSocket();
      }
      yield _sseEvent({'type': 'done'});
    } catch (e, st) {
      stderr.writeln('[stream] $e\n$st');
      yield _sseEvent({'type': 'error', 'message': e.toString()});
    }
  }

  /// SSE wire format: `data: <json>\n\n`. JSON encoding handles escaping
  /// of newlines / quotes inside the payload.
  List<int> _sseEvent(Map<String, dynamic> data) =>
      utf8.encode('data: ${jsonEncode(data)}\n\n');

  Future<void> _yieldToSocket() =>
      Future<void>.delayed(const Duration(milliseconds: 12));

  // ----------------------------- Prompt + citations -----------------------------

  String _buildPrompt(String query, List<ScoredChunk> hits) {
    final ctx = StringBuffer();
    for (var i = 0; i < hits.length; i++) {
      final h = hits[i];
      ctx.writeln('[${i + 1}] (${h.chunk.docName} #${h.chunk.chunkIndex})');
      ctx.writeln(h.chunk.text);
      ctx.writeln();
    }
    return '''
你是一位嚴謹的閱讀助手。請只依據下方文件片段回答問題。
- 若資料不足，明確回答「資料中沒有提到」。
- 回答中以 [1]、[2] 等標註所引用的片段。

### 文件片段
$ctx
### 問題
$query

### 回答
''';
  }

  List<Map<String, dynamic>> _citations(List<ScoredChunk> hits) {
    return hits
        .map<Map<String, dynamic>>((h) => {
              'doc': h.chunk.docName,
              'chunkIndex': h.chunk.chunkIndex,
              'score': h.score,
              'snippet': _truncate(h.chunk.text, 240),
            })
        .toList();
  }

  String _truncate(String s, int n) =>
      s.length <= n ? s : '${s.substring(0, n)}…';

  Response _jsonResponse(Map<String, dynamic> data, {int statusCode = 200}) {
    return Response(
      statusCode,
      body: jsonEncode(data),
      headers: _jsonHeaders,
    );
  }

  // ----------------------------- Lifecycle -----------------------------

  /// Starts the HTTP server.
  ///
  /// Safe by default: binds to `127.0.0.1` only, so nothing on the LAN
  /// can reach the server unless [lanMode] is explicitly true.
  ///
  /// [lanMode] = true binds to all IPv4 addresses (`0.0.0.0`) so a phone
  /// on the same Wi-Fi can connect. **Requires [authToken] to be set** —
  /// otherwise [start] throws [StateError] rather than expose an
  /// unauthenticated server to the network.
  Future<HttpServer> start({
    int port = 8080,
    bool lanMode = false,
  }) async {
    if (lanMode && (authToken == null || authToken!.trim().isEmpty)) {
      throw StateError(
        'LAN mode requires authToken. Refusing to start an open server '
        'reachable from the network. Set AI_LIB_TOKEN to a non-blank value, '
        'or call start() without lanMode: true to bind loopback only.',
      );
    }

    final cors = corsHeaders(
      headers: {
        'Access-Control-Allow-Origin': '*',
        'Access-Control-Allow-Methods': 'GET, POST, OPTIONS',
        'Access-Control-Allow-Headers': 'Origin, Content-Type, Authorization',
      },
    );

    // Order matters: CORS must wrap the error handler so error responses
    // still carry CORS headers.
    final handler = const Pipeline()
        .addMiddleware(logRequests())
        .addMiddleware(cors)
        .addMiddleware(_errorHandler())
        .addHandler(_router.call);

    final bindAddress =
        lanMode ? InternetAddress.anyIPv4 : InternetAddress.loopbackIPv4;
    final server = await io.serve(handler, bindAddress, port);
    final actualPort = server.port; // resolves 0 → real OS-assigned port
    final hostLabel = lanMode ? '0.0.0.0' : '127.0.0.1';

    stdout.writeln('Local AI Server running on http://$hostLabel:$actualPort');
    if (lanMode) {
      stdout.writeln('LAN mode enabled with Bearer auth.');
      stdout.writeln('Local IPs (give one of these to your phone):');
      final ips = await _localIps();
      if (ips.isEmpty) {
        stdout.writeln('  (none detected — check your network)');
      } else {
        for (final entry in ips) {
          stdout.writeln(
              '  http://${entry.address}:$actualPort  (${entry.iface})');
        }
      }
    } else {
      stdout.writeln('Loopback-only. Pass lanMode: true + AI_LIB_TOKEN '
          'to expose to the LAN.');
    }
    if (authToken == null || authToken!.isEmpty) {
      stdout.writeln('Auth: OFF (loopback only — fine for local dev).');
    } else {
      stdout.writeln('Auth: Bearer ******** (token configured)');
    }
    stdout.writeln('Bound to ${server.address.address}:$actualPort');
    return server;
  }

  Middleware _errorHandler() {
    return (Handler innerHandler) {
      return (Request req) async {
        try {
          return await innerHandler(req);
        } on FormatException catch (e) {
          return _jsonResponse({'error': e.message}, statusCode: 400);
        } catch (e, st) {
          stderr.writeln('[unhandled] $e\n$st');
          return _jsonResponse(
            {'error': 'Internal server error'},
            statusCode: 500,
          );
        }
      };
    };
  }

  /// Returns all non-loopback IPv4 addresses with interface names.
  /// Caller picks the right one — avoids guessing wrong when Docker, VPN,
  /// or other virtual adapters are present.
  Future<List<({String iface, String address})>> _localIps() async {
    try {
      final out = <({String iface, String address})>[];
      for (final iface in await NetworkInterface.list()) {
        for (final addr in iface.addresses) {
          if (addr.type == InternetAddressType.IPv4 && !addr.isLoopback) {
            out.add((iface: iface.name, address: addr.address));
          }
        }
      }
      return out;
    } catch (_) {
      return const [];
    }
  }
}

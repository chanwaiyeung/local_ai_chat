import 'api_client.dart';

class LanguageLearningService {
  LanguageLearningService({required ReaderApi api}) : _api = api;

  final ReaderApi _api;

  Future<String> explainWord({
    required String word,
    required String context,
    String? docName,
    String targetLanguage = '繁體中文',
  }) async {
    final result = await _api.query(
      query: buildExplainPrompt(
        word: word,
        context: context,
        targetLanguage: targetLanguage,
      ),
      docName: docName,
    );
    return result['answer'] as String? ?? '無法取得解釋';
  }

  String buildExplainPrompt({
    required String word,
    required String context,
    String targetLanguage = '繁體中文',
  }) {
    return '請用$targetLanguage簡短解釋「$word」在下列上下文中的意思，'
        '並提供 1 個自然例句。\n\n上下文：$context';
  }
}

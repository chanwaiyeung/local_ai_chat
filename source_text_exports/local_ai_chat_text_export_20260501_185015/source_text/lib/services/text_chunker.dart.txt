// lib/services/text_chunker.dart

import 'vector_store.dart';

class TextChunker {
  static List<DocChunk> split(
    String text, {
    required String docName,
    int maxChars = 800,
    int overlap = 0,
  }) {
    if (maxChars <= 0) {
      throw ArgumentError.value(maxChars, 'maxChars', 'must be positive');
    }
    if (overlap < 0) {
      throw ArgumentError.value(overlap, 'overlap', 'must be non-negative');
    }
    if (overlap >= maxChars) {
      throw ArgumentError.value(
        overlap,
        'overlap',
        'must be smaller than maxChars',
      );
    }

    final paragraphs = text
        .replaceAll('\r', '')
        .split(RegExp(r'\n\s*\n'))
        .map((part) => part.trim())
        .where((part) => part.isNotEmpty)
        .toList();

    final chunks = <String>[];
    for (final paragraph in paragraphs) {
      chunks.addAll(_splitParagraph(paragraph, maxChars));
    }

    final withOverlap = <String>[];
    for (final chunk in chunks) {
      if (overlap > 0 && withOverlap.isNotEmpty) {
        final previous = withOverlap.last;
        final tail = previous.length <= overlap
            ? previous
            : previous.substring(previous.length - overlap);
        withOverlap.add('$tail\n…\n$chunk');
      } else {
        withOverlap.add(chunk);
      }
    }

    return [
      for (var i = 0; i < withOverlap.length; i++)
        DocChunk(docName: docName, chunkIndex: i, text: withOverlap[i]),
    ];
  }

  static List<String> _splitParagraph(String paragraph, int maxChars) {
    if (paragraph.length <= maxChars) return [paragraph];

    final sentences = paragraph
        .split(RegExp(r'(?<=[。！？.!?])\s+|(?<=[。！？])(?=\S)'))
        .where((sentence) => sentence.trim().isNotEmpty)
        .map((sentence) => sentence.trim())
        .toList();

    if (sentences.length == 1 && sentences.single.length > maxChars) {
      return _hardSplit(sentences.single, maxChars);
    }

    final chunks = <String>[];
    final buffer = StringBuffer();
    for (final sentence in sentences) {
      if (sentence.length > maxChars) {
        if (buffer.isNotEmpty) {
          chunks.add(buffer.toString().trim());
          buffer.clear();
        }
        chunks.addAll(_hardSplit(sentence, maxChars));
        continue;
      }
      final separator = buffer.isEmpty ? '' : ' ';
      if (buffer.length + separator.length + sentence.length > maxChars) {
        chunks.add(buffer.toString().trim());
        buffer.clear();
      }
      if (buffer.isNotEmpty) buffer.write(' ');
      buffer.write(sentence);
    }
    if (buffer.isNotEmpty) chunks.add(buffer.toString().trim());
    return chunks;
  }

  static List<String> _hardSplit(String text, int maxChars) {
    final chunks = <String>[];
    for (var i = 0; i < text.length; i += maxChars) {
      final end = (i + maxChars).clamp(0, text.length);
      chunks.add(text.substring(i, end));
    }
    return chunks;
  }
}

# tool/install_android.ps1
#
param(
  [switch] $InstallSdk,
  [switch] $Launch,
  [switch] $SkipAvd,
  [string] $AvdName = 'android_emulator',
  [string] $ApiLevel = '36',
  [string] $Device = 'pixel_6',
  [string] $SdkRoot = ''
)

# Android setup helper for this Flutter project.
# - Ensures android/ platform scaffolding exists.
# - Detects or installs Android command-line tools when -InstallSdk is passed.
# - Installs SDK packages and creates an AVD when sdkmanager/avdmanager are available.

$ErrorActionPreference = 'Stop'

$root = Resolve-Path (Join-Path $PSScriptRoot '..')
Set-Location $root

function Add-ToPath {
  param([string] $Path)
  if ($Path -and (Test-Path $Path) -and ($env:Path -notlike "*$Path*")) {
    $env:Path = $Path + [IO.Path]::PathSeparator + $env:Path
  }
}

function Find-FirstExisting {
  param([string[]] $Paths)
  foreach ($path in $Paths) {
    if ($path -and (Test-Path $path)) {
      return (Resolve-Path $path).Path
    }
  }
  return $null
}

function Invoke-Checked {
  param(
    [Parameter(Mandatory = $true)]
    [string] $Label,
    [Parameter(Mandatory = $true)]
    [scriptblock] $Command
  )

  Write-Host ''
  Write-Host "==> $Label" -ForegroundColor Cyan
  & $Command
  if ($LASTEXITCODE -ne 0) {
    throw "$Label failed with exit code $LASTEXITCODE"
  }
}

$flutterHome = if ($env:FLUTTER_HOME) { $env:FLUTTER_HOME } else { Join-Path $HOME 'flutter' }
Add-ToPath (Join-Path $flutterHome 'bin')

$javaHome = if ($env:JAVA_HOME) {
  $env:JAVA_HOME
} else {
  Find-FirstExisting @(
    'C:\Program Files\Android\openjdk\jdk-21.0.8',
    'C:\Program Files\Android\openjdk\jdk-21.0.7',
    'C:\Program Files\Android\openjdk\jdk-21',
    'C:\Program Files\Android Studio\jbr'
  )
}
if ($javaHome) {
  $env:JAVA_HOME = $javaHome
  Add-ToPath (Join-Path $javaHome 'bin')
}

if (-not (Get-Command flutter -ErrorAction SilentlyContinue)) {
  throw 'flutter is not on PATH. Run tool\install_flutter.cmd first.'
}

if (-not (Test-Path (Join-Path $root 'android'))) {
  Invoke-Checked 'flutter create --platforms=android,windows .' {
    flutter create --platforms=android,windows .
  }
} else {
  Write-Host 'android/ scaffold already exists.' -ForegroundColor Green
}

if (-not $SdkRoot) {
  $SdkRoot = if ($env:ANDROID_SDK_ROOT) {
    $env:ANDROID_SDK_ROOT
} elseif ($env:ANDROID_HOME) {
    $env:ANDROID_HOME
  } elseif (Test-Path 'C:\Android\Sdk') {
    'C:\Android\Sdk'
  } else {
    Join-Path $env:LOCALAPPDATA 'Android\Sdk'
  }
}

New-Item -ItemType Directory -Force -Path $SdkRoot | Out-Null
$env:ANDROID_SDK_ROOT = $SdkRoot
$env:ANDROID_HOME = $SdkRoot
Add-ToPath (Join-Path $SdkRoot 'platform-tools')
Add-ToPath (Join-Path $SdkRoot 'emulator')
Add-ToPath (Join-Path $SdkRoot 'cmdline-tools\latest\bin')

$sdkManager = Get-Command sdkmanager -ErrorAction SilentlyContinue
if (-not $sdkManager) {
  $candidate = Join-Path $SdkRoot 'cmdline-tools\latest\bin\sdkmanager.bat'
  if (Test-Path $candidate) {
    $sdkManager = Get-Item $candidate
  }
}

if (-not $sdkManager -and $InstallSdk) {
  $downloadUrl = 'https://dl.google.com/android/repository/commandlinetools-win-14742923_latest.zip'
  $downloadDir = Join-Path $root 'build\android-sdk'
  $zipPath = Join-Path $downloadDir 'commandlinetools-win_latest.zip'
  $extractDir = Join-Path $downloadDir 'cmdline-tools-extract'
  $latestDir = Join-Path $SdkRoot 'cmdline-tools\latest'

  New-Item -ItemType Directory -Force -Path $downloadDir | Out-Null
  Remove-Item -LiteralPath $extractDir -Recurse -Force -ErrorAction SilentlyContinue
  Write-Host "Downloading Android command-line tools from $downloadUrl" -ForegroundColor Yellow
  Invoke-WebRequest -Uri $downloadUrl -OutFile $zipPath
  Expand-Archive -LiteralPath $zipPath -DestinationPath $extractDir -Force

  Remove-Item -LiteralPath $latestDir -Recurse -Force -ErrorAction SilentlyContinue
  New-Item -ItemType Directory -Force -Path (Split-Path $latestDir) | Out-Null
  Move-Item -LiteralPath (Join-Path $extractDir 'cmdline-tools') -Destination $latestDir
  Add-ToPath (Join-Path $latestDir 'bin')
  $sdkManager = Get-Item (Join-Path $latestDir 'bin\sdkmanager.bat')
}

if (-not $sdkManager) {
  Write-Host ''
  Write-Host 'Android SDK command-line tools are missing.' -ForegroundColor Yellow
  Write-Host "SDK root checked: $SdkRoot"
  Write-Host 'Run this helper with SDK installation enabled:'
  Write-Host ''
  Write-Host '  powershell -ExecutionPolicy Bypass -File tool\install_android.ps1 -InstallSdk'
  Write-Host ''
  Write-Host 'Or install Android Studio, then rerun this script.'
  exit 2
}

$sdkManagerPath = $sdkManager.Source
if (-not $sdkManagerPath) { $sdkManagerPath = $sdkManager.FullName }
$avdManagerPath = Join-Path (Split-Path $sdkManagerPath) 'avdmanager.bat'
if (-not (Test-Path $avdManagerPath)) {
  throw "avdmanager.bat not found next to sdkmanager: $avdManagerPath"
}

$packages = @(
  'platform-tools',
  'emulator',
  "platforms;android-$ApiLevel",
  'build-tools;28.0.3',
  'build-tools;35.0.0',
  'cmdline-tools;latest'
)

$systemImage = "system-images;android-$ApiLevel;google_apis;x86_64"
if (-not $SkipAvd) {
  $freeBytes = (Get-PSDrive -Name ((Split-Path -Qualifier $SdkRoot).TrimEnd(':'))).Free
  $minimumFreeBytes = 8GB
  if ($freeBytes -lt $minimumFreeBytes) {
    $freeGb = [Math]::Round($freeBytes / 1GB, 2)
    throw "Only $freeGb GB free on the SDK drive. Free at least 8 GB for the emulator system image, or rerun with -SkipAvd to set up Android for a physical device only."
  }
  $packages += $systemImage
}

Write-Host ''
Write-Host "Android SDK root: $SdkRoot" -ForegroundColor Green
Write-Host "Java home: $env:JAVA_HOME" -ForegroundColor Green

Invoke-Checked 'sdkmanager --licenses' {
  $licenseInput = Join-Path $root 'build\android-sdk\license-input.txt'
  New-Item -ItemType Directory -Force -Path (Split-Path $licenseInput) | Out-Null
  Set-Content -LiteralPath $licenseInput -Value ((('y' + [Environment]::NewLine) * 200)) -Encoding ascii
  cmd /c "`"$sdkManagerPath`" --sdk_root=`"$SdkRoot`" --licenses < `"$licenseInput`""
}

Invoke-Checked 'sdkmanager packages' {
  & $sdkManagerPath --sdk_root=$SdkRoot @packages
}

Invoke-Checked 'flutter config --android-sdk' {
  flutter config --android-sdk $SdkRoot
}

if ($SkipAvd) {
  Write-Host 'Skipping AVD creation because -SkipAvd was supplied.' -ForegroundColor Yellow
} else {
  $existingAvds = & $avdManagerPath list avd
  if ($existingAvds -match [Regex]::Escape("Name: $AvdName")) {
    Write-Host "AVD '$AvdName' already exists." -ForegroundColor Green
  } else {
    Invoke-Checked "avdmanager create avd $AvdName" {
      'no' | & $avdManagerPath create avd `
        --name $AvdName `
        --package $systemImage `
        --device $Device `
        --force
    }
  }
}

Invoke-Checked 'flutter doctor -v' {
  flutter doctor -v
}

Invoke-Checked 'flutter emulators' {
  flutter emulators
}

if ($Launch) {
  $emulatorExe = Join-Path $SdkRoot 'emulator\emulator.exe'
  if (-not (Test-Path $emulatorExe)) {
    throw "emulator.exe not found: $emulatorExe"
  }

  Write-Host ''
  Write-Host "Launching $AvdName..." -ForegroundColor Cyan
  Start-Process -FilePath $emulatorExe -ArgumentList @('-avd', $AvdName) -WindowStyle Hidden
}

Write-Host ''
Write-Host 'Android setup helper completed.' -ForegroundColor Green

// bin/index.dart
//
// CLI to build / extend the RAG index. Walks the given files / dirs, finds
// every .txt, splits + embeds + appends to the NDJSON vector store.
//
// Usage:
//   dart run bin/index.dart books/                 # index a folder
//   dart run bin/index.dart book.txt other.txt     # index specific files
//
// Env vars:
//   AI_LIB_INDEX   NDJSON path (default: data/vectors.ndjson)
//   EMBED_MODEL    Ollama embed model (default: bge-m3)
//   OLLAMA_URL     Ollama base URL (default: http://localhost:11434)

import 'dart:io';

import 'package:local_ai_chat/services/document_loader.dart';
import 'package:local_ai_chat/services/embedding_service.dart';
import 'package:local_ai_chat/services/rag_service.dart';
import 'package:local_ai_chat/services/vector_store.dart';
import 'package:path/path.dart' as p;

Future<void> main(List<String> args) async {
  if (args.isEmpty) {
    stderr.writeln('Usage: dart run bin/index.dart <file-or-dir> [<more>...]');
    exit(64);
  }

  final indexPath =
      Platform.environment['AI_LIB_INDEX'] ?? 'data/vectors.ndjson';
  final embedModel = Platform.environment['EMBED_MODEL'] ?? 'bge-m3';
  final ollamaUrl =
      Platform.environment['OLLAMA_URL'] ?? 'http://localhost:11434';

  stdout.writeln('Index path:   $indexPath');
  stdout.writeln('Embed model:  $embedModel');
  stdout.writeln('Ollama URL:   $ollamaUrl');
  stdout.writeln('');

  final store = VectorStore();
  await store.load();
  stdout.writeln(
      'Loaded ${store.length} existing chunks across ${store.docNames.length} docs.');

  final embedder = EmbeddingService(baseUrl: ollamaUrl, model: embedModel);
  final rag = RagService(embedder: embedder, store: store);

  final files = <String>[];
  for (final arg in args) {
    final type = FileSystemEntity.typeSync(arg);
    if (type == FileSystemEntityType.file) {
      if (isSupportedDocument(arg)) {
        files.add(arg);
      } else {
        stderr.writeln(
            'Skipping (unsupported extension): $arg — supported: ${supportedExtensions.join(", ")}');
      }
    } else if (type == FileSystemEntityType.directory) {
      await for (final f in Directory(arg).list(recursive: true)) {
        if (f is File && isSupportedDocument(f.path)) {
          files.add(f.path);
        }
      }
    } else {
      stderr.writeln('Skipping (not file or dir): $arg');
    }
  }

  if (files.isEmpty) {
    stderr.writeln(
        'No supported files found. Supported: ${supportedExtensions.join(", ")}');
    embedder.close();
    exit(1);
  }

  stdout.writeln('Indexing ${files.length} file(s)...');
  for (final f in files) {
    stdout.write('  ${p.basename(f)}: ');
    try {
      final n = await rag.indexFile(f);
      stdout.writeln('$n chunks');
    } catch (e) {
      stdout.writeln('FAILED ($e)');
    }
  }

  stdout.writeln('');
  stdout.writeln(
      'Done. Index now has ${store.length} chunks across ${store.docNames.length} docs.');
  embedder.close();
}

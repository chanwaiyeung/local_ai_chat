// lib/services/document_loader.dart
//
// Reads a file from disk and returns plain text, ready for the chunker.
// Dispatches by extension. Adding a format = adding one case.
//
// Currently handled:
//   .txt / .md / .markdown        →  read as UTF-8
//   .epub                         →  parse with epubx, walk chapters, HTML → text
//   .pdf                          →  shell out to `pdftotext` (Poppler) on PATH
//   .png / .jpg / .jpeg / .webp   →  shell out to `tesseract` (Tesseract) on PATH
//   .cbz                          →  unzip + per-image OCR via `tesseract`
//
// OCR languages: set $OCR_LANGS env var (default: chi_tra+eng).
// Anything else throws FormatException.

import 'dart:convert';
import 'dart:io';

import 'package:archive/archive.dart';
import 'package:epubx/epubx.dart' as epub;
import 'package:html/dom.dart' as dom;
import 'package:html/parser.dart' as html_parser;
import 'package:path/path.dart' as p;

const List<String> supportedExtensions = [
  '.txt',
  '.md',
  '.markdown',
  '.epub',
  '.pdf',
  '.cbz',
  '.png',
  '.jpg',
  '.jpeg',
  '.webp',
];

const List<String> _imageExtensions = ['.png', '.jpg', '.jpeg', '.webp'];

bool isSupportedDocument(String path) {
  final ext = p.extension(path).toLowerCase();
  return supportedExtensions.contains(ext);
}

/// Loads [path] as plain text. Throws [FormatException] if the format isn't
/// supported or the file isn't a parseable instance of that format.
Future<String> loadDocument(String path) async {
  final ext = p.extension(path).toLowerCase();
  switch (ext) {
    case '.txt':
    case '.md':
    case '.markdown':
      return File(path).readAsString();
    case '.epub':
      return _loadEpub(path);
    case '.pdf':
      return _loadPdf(path);
    case '.cbz':
      return _loadCbz(path);
    case '.png':
    case '.jpg':
    case '.jpeg':
    case '.webp':
      return _loadImage(path);
    default:
      throw FormatException(
          'Unsupported file type: $ext (supported: ${supportedExtensions.join(", ")})');
  }
}

// ----------------------------- EPUB -----------------------------

Future<String> _loadEpub(String path) async {
  final bytes = await File(path).readAsBytes();
  final book = await epub.EpubReader.readBook(bytes);
  final buf = StringBuffer();

  // Title page.
  final title = book.Title;
  if (title != null && title.trim().isNotEmpty) {
    buf.writeln('# $title');
    buf.writeln();
  }
  final author = book.Author;
  if (author != null && author.trim().isNotEmpty) {
    buf.writeln('by $author');
    buf.writeln();
  }

  for (final chapter in book.Chapters ?? const <epub.EpubChapter>[]) {
    _appendChapter(chapter, buf);
  }

  return buf.toString().trim();
}

void _appendChapter(epub.EpubChapter chapter, StringBuffer buf) {
  final title = chapter.Title;
  if (title != null && title.trim().isNotEmpty) {
    buf.writeln('## ${title.trim()}');
    buf.writeln();
  }
  final html = chapter.HtmlContent;
  if (html != null && html.trim().isNotEmpty) {
    final text = htmlToText(html);
    if (text.isNotEmpty) {
      buf.writeln(text);
      buf.writeln();
    }
  }
  for (final sub in chapter.SubChapters ?? const <epub.EpubChapter>[]) {
    _appendChapter(sub, buf);
  }
}

/// Converts HTML to a paragraph-preserving plain-text representation.
/// Block-level closing tags become `\n\n` so the chunker can split sensibly.
String htmlToText(String html) {
  final doc = html_parser.parse(html);
  final root = doc.body ?? doc.documentElement;
  if (root == null) return '';

  final buf = StringBuffer();
  _collectText(root, buf);

  return buf
      .toString()
      .replaceAll(RegExp(r'[ \t]+'), ' ')
      .replaceAll(RegExp(r' *\n *'), '\n')
      .replaceAll(RegExp(r'\n{3,}'), '\n\n')
      .trim();
}

const _blockTags = <String>{
  'p',
  'div',
  'h1',
  'h2',
  'h3',
  'h4',
  'h5',
  'h6',
  'li',
  'blockquote',
  'pre',
  'tr',
  'section',
  'article',
  'aside',
  'figure',
  'figcaption',
};

void _collectText(dom.Node node, StringBuffer buf) {
  if (node is dom.Text) {
    buf.write(node.text);
    return;
  }
  if (node is dom.Element) {
    final tag = node.localName?.toLowerCase();
    if (tag == 'br') {
      buf.write('\n');
      return;
    }
    if (tag == 'script' || tag == 'style') {
      return; // skip non-content
    }
    final block = tag != null && _blockTags.contains(tag);
    for (final child in node.nodes) {
      _collectText(child, buf);
    }
    if (block) buf.write('\n\n');
  }
}

// ----------------------------- PDF -----------------------------

/// Best-effort PDF text extraction via `pdftotext` (Poppler). Throws a
/// helpful error if the binary isn't on PATH.
Future<String> _loadPdf(String path) async {
  try {
    final result = await Process.run(
      'pdftotext',
      ['-layout', '-enc', 'UTF-8', path, '-'],
      stdoutEncoding: utf8,
      stderrEncoding: utf8,
    );
    if (result.exitCode != 0) {
      throw FormatException(
          'pdftotext exited ${result.exitCode}: ${result.stderr}');
    }
    return result.stdout as String;
  } on ProcessException {
    throw const FormatException(
      'PDF support needs `pdftotext` on PATH. Install Poppler:\n'
      '  Windows: scoop install poppler  (or https://github.com/oschwartz10612/poppler-windows)\n'
      '  macOS:   brew install poppler\n'
      '  Linux:   sudo apt install poppler-utils\n'
      'Or convert the PDF to .txt manually and re-index.',
    );
  }
}

// ----------------------------- Image / CBZ via Tesseract -----------------------------

/// Single-image OCR. Useful for «snap a photo of a book page, ask AI».
Future<String> _loadImage(String path) async {
  return _ocrFile(path);
}

/// Comic Book ZIP: extract images, OCR each in natural reading order,
/// concatenate with page-number headings.
Future<String> _loadCbz(String path) async {
  final bytes = await File(path).readAsBytes();
  final archive = ZipDecoder().decodeBytes(bytes);

  final images = archive
      .where((f) =>
          f.isFile &&
          _imageExtensions.contains(p.extension(f.name).toLowerCase()))
      .toList()
    ..sort((a, b) => _naturalCompare(a.name, b.name));

  if (images.isEmpty) {
    throw const FormatException(
        'CBZ contains no recognised image files (.png/.jpg/.jpeg/.webp).');
  }

  final tempDir = await Directory.systemTemp.createTemp('cbz-ocr-');
  final buf = StringBuffer();
  try {
    for (var i = 0; i < images.length; i++) {
      final entry = images[i];
      final imgFile = File(p.join(tempDir.path, p.basename(entry.name)));
      await imgFile.writeAsBytes(entry.content as List<int>);

      stderr.writeln(
          '  OCR ${i + 1}/${images.length}: ${p.basename(entry.name)}');
      final text = await _ocrFile(imgFile.path);
      final trimmed = text.trim();
      if (trimmed.isEmpty) continue;

      buf.writeln('## ${p.basenameWithoutExtension(entry.name)}');
      buf.writeln();
      buf.writeln(trimmed);
      buf.writeln();
    }
  } finally {
    if (await tempDir.exists()) {
      try {
        await tempDir.delete(recursive: true);
      } catch (_) {/* swallow cleanup races */}
    }
  }

  return buf.toString().trim();
}

/// Runs `tesseract <img> stdout -l <langs> --psm 6`. Throws a helpful
/// install-instruction error if tesseract isn't on PATH.
Future<String> _ocrFile(String imagePath) async {
  final langs = Platform.environment['OCR_LANGS'] ?? 'chi_tra+eng';
  try {
    final result = await Process.run(
      'tesseract',
      [imagePath, 'stdout', '-l', langs, '--psm', '6'],
      stdoutEncoding: utf8,
      stderrEncoding: utf8,
    );
    if (result.exitCode != 0) {
      final err = (result.stderr as String).trim();
      throw FormatException(
          'tesseract exited ${result.exitCode}: $err\n'
          'If err mentions "Failed loading language", install missing trained data:\n'
          '  https://github.com/tesseract-ocr/tessdata/  (drop into your tessdata dir)\n'
          'Or set OCR_LANGS to languages you have, e.g. OCR_LANGS=eng');
    }
    return result.stdout as String;
  } on ProcessException {
    throw const FormatException(
      'Image / CBZ OCR needs `tesseract` on PATH. Install Tesseract:\n'
      '  Windows: winget install --id UB-Mannheim.TesseractOCR\n'
      '           (or scoop install tesseract)\n'
      '  macOS:   brew install tesseract tesseract-lang\n'
      '  Linux:   sudo apt install tesseract-ocr tesseract-ocr-chi-tra\n'
      'Set OCR_LANGS env to override languages (default: chi_tra+eng).',
    );
  }
}

/// Compare two filenames so that `2.jpg` < `10.jpg`. Standard string sort
/// would put `10.jpg` before `2.jpg`, which scrambles unpadded CBZ pages.
int _naturalCompare(String a, String b) {
  final tokenizer = RegExp(r'\d+|\D+');
  final ar = tokenizer.allMatches(a).map((m) => m[0]!).toList();
  final br = tokenizer.allMatches(b).map((m) => m[0]!).toList();
  final n = ar.length < br.length ? ar.length : br.length;
  for (var i = 0; i < n; i++) {
    final ap = ar[i];
    final bp = br[i];
    final aNum = int.tryParse(ap);
    final bNum = int.tryParse(bp);
    if (aNum != null && bNum != null) {
      final cmp = aNum.compareTo(bNum);
      if (cmp != 0) return cmp;
    } else {
      final cmp = ap.compareTo(bp);
      if (cmp != 0) return cmp;
    }
  }
  return ar.length.compareTo(br.length);
}

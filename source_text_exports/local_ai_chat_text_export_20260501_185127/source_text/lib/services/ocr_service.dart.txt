// lib/services/ocr_service.dart
//
// Mobile / desktop image OCR. On native platforms (Windows / macOS / Linux /
// Android / iOS) this shells out to `tesseract`, the same engine used by
// the server-side bin/index.dart pipeline. On Web we fall back to an
// optional sidecar file (.txt) or throw — there is no in-browser OCR yet.
//
// Tests use a hand-rolled `Fake extends OcrService` to bypass the native
// path entirely, so this class doesn't need a separate "fake mode" flag.

import 'dart:convert';
import 'dart:io';

import 'package:archive/archive_io.dart';
import 'package:flutter/foundation.dart';

class OcrService {
  OcrService({this.languages, this.tesseractPath = 'tesseract'});

  /// Plus-separated tesseract language codes (e.g. 'chi_tra+eng').
  /// If null, reads `OCR_LANGS` env (default 'chi_tra+eng' on native;
  /// unused on web).
  final String? languages;
  final String tesseractPath;

  String get _langs {
    if (languages != null && languages!.isNotEmpty) return languages!;
    if (kIsWeb) return 'eng';
    return Platform.environment['OCR_LANGS'] ?? 'chi_tra+eng';
  }

  /// Extracts text from a single image. On native platforms this runs
  /// `tesseract <path> stdout -l <langs> --psm 6`. On Web we look for an
  /// optional sidecar .txt next to the image; if neither works, throws.
  Future<String> extractTextFromImage(String imagePath) async {
    if (kIsWeb) {
      // Web cannot exec processes. Allow callers to ship a precomputed
      // sidecar text file for testing / demos.
      throw UnsupportedError(
        'In-browser OCR is not implemented. Use the desktop / mobile app '
        '(which shells out to tesseract), or pre-OCR with '
        '`dart run bin/index.dart` and ask via /query instead.',
      );
    }

    final image = File(imagePath);
    if (!await image.exists()) {
      throw FileSystemException('Image not found', imagePath);
    }

    // Honour an optional .txt sidecar so users can hand-correct OCR.
    // (Same pattern Calibre / Plex use for stubborn files.)
    final sidecar = File('$imagePath.txt');
    if (await sidecar.exists()) {
      final text = (await sidecar.readAsString()).trim();
      if (text.isNotEmpty) return text;
    }

    return _runTesseract(imagePath);
  }

  /// Extracts text from a single page of a CBZ. Native only — uses a temp
  /// directory for the unpacked image, runs tesseract, cleans up.
  Future<String> extractTextFromCbzPage(String cbzPath, int pageIndex) async {
    if (kIsWeb) {
      throw UnsupportedError(
        'CBZ OCR is not available on web. Use `dart run bin/index.dart` '
        'on a desktop with tesseract installed.',
      );
    }

    final file = File(cbzPath);
    if (!await file.exists()) {
      throw FileSystemException('CBZ not found', cbzPath);
    }

    final archive = ZipDecoder().decodeBytes(await file.readAsBytes());
    final images = archive.files
        .where((entry) => entry.isFile && _isImageName(entry.name))
        .toList()
      ..sort((a, b) => _naturalCompare(a.name, b.name));

    if (pageIndex < 0 || pageIndex >= images.length) {
      throw RangeError.range(pageIndex, 0, images.length - 1, 'pageIndex');
    }

    final tempDir = await Directory.systemTemp.createTemp('cbz-ocr-');
    try {
      final entry = images[pageIndex];
      final imgFile = File('${tempDir.path}/${_sanitize(entry.name)}');
      await imgFile.writeAsBytes(entry.content as List<int>);
      return _runTesseract(imgFile.path);
    } finally {
      try {
        await tempDir.delete(recursive: true);
      } catch (_) {
        // Best-effort cleanup.
      }
    }
  }

  void dispose() {}

  // --------------------------- Internals ---------------------------

  Future<String> _runTesseract(String imagePath) async {
    try {
      final result = await Process.run(
        tesseractPath,
        [imagePath, 'stdout', '-l', _langs, '--psm', '6'],
        stdoutEncoding: utf8,
        stderrEncoding: utf8,
      );
      if (result.exitCode != 0) {
        final err = (result.stderr as String).trim();
        throw Exception('tesseract exited ${result.exitCode}: $err');
      }
      return (result.stdout as String).trim();
    } on ProcessException {
      throw Exception(
        'tesseract not on PATH. Run tool/install_tesseract.cmd '
        '(Windows) or brew/apt install tesseract.',
      );
    }
  }

  bool _isImageName(String name) {
    final lower = name.toLowerCase();
    return lower.endsWith('.jpg') ||
        lower.endsWith('.jpeg') ||
        lower.endsWith('.png') ||
        lower.endsWith('.webp');
  }

  /// Strips any leading directory components so we can drop the file flat
  /// into the temp dir without recreating CBZ folder structure.
  String _sanitize(String name) {
    final i = name.lastIndexOf(RegExp(r'[/\\]'));
    return i < 0 ? name : name.substring(i + 1);
  }

  /// `2.jpg` < `10.jpg`. Standard string sort puts `10.jpg` first.
  int _naturalCompare(String a, String b) {
    final tokenizer = RegExp(r'\d+|\D+');
    final ar = tokenizer.allMatches(a).map((m) => m[0]!).toList();
    final br = tokenizer.allMatches(b).map((m) => m[0]!).toList();
    final n = ar.length < br.length ? ar.length : br.length;
    for (var i = 0; i < n; i++) {
      final ap = ar[i];
      final bp = br[i];
      final aNum = int.tryParse(ap);
      final bNum = int.tryParse(bp);
      if (aNum != null && bNum != null) {
        final cmp = aNum.compareTo(bNum);
        if (cmp != 0) return cmp;
      } else {
        final cmp = ap.compareTo(bp);
        if (cmp != 0) return cmp;
      }
    }
    return ar.length.compareTo(br.length);
  }
}

// lib/controllers/health_controller.dart
//
// Phase 6.7 (v2.3) — Health module controller.
// Storage: VectorStore Collection 'Health', empty embeddings (Strategy C).
// Mirrors ContactController / ExpenseController.

import 'dart:math' as math;

import 'package:flutter/foundation.dart';

import '../models/health_record.dart';
import '../services/vector_store.dart';

class HealthController extends ChangeNotifier {
  HealthController(this._store);

  static const String collectionName = 'Health';
  static const String _idPrefix = 'health_';

  final VectorStore _store;
  final math.Random _rng = math.Random();

  // ------------------------------------------------------------------ reads

  /// All records, sorted by measurement date (newest first).
  List<HealthRecord> getAllRecords() {
    final out = <HealthRecord>[];
    for (final c in _store.chunksInCollection(collectionName)) {
      try {
        out.add(HealthRecord.decode(c.text));
      } catch (_) {
        // skip rather than throw
      }
    }
    out.sort((a, b) => b.date.compareTo(a.date));
    return out;
  }

  HealthRecord? findById(String id) {
    for (final r in getAllRecords()) {
      if (r.id == id) return r;
    }
    return null;
  }

  /// Keyword search over [HealthRecord.toSearchText]. Empty query → all.
  List<HealthRecord> searchRecords(String query) {
    final q = query.trim().toLowerCase();
    if (q.isEmpty) return getAllRecords();
    return getAllRecords()
        .where((r) => r.toSearchText().toLowerCase().contains(q))
        .toList();
  }

  /// Records whose [HealthRecord.date] falls within [start, end] inclusive.
  List<HealthRecord> getRecordsInRange(DateTime start, DateTime end) {
    final lo = DateTime(start.year, start.month, start.day);
    final hi = DateTime(end.year, end.month, end.day, 23, 59, 59);
    return getAllRecords()
        .where((r) =>
            !r.date.isBefore(lo) && !r.date.isAfter(hi))
        .toList();
  }

  /// Records from the last [days] days (inclusive of today).
  List<HealthRecord> getRecordsLastNDays(int days) {
    final now = DateTime.now();
    final start = DateTime(now.year, now.month, now.day)
        .subtract(Duration(days: days - 1));
    return getRecordsInRange(start, now);
  }

  int get count => _store.chunksInCollection(collectionName).length;

  /// Aggregated stats over a chosen window. Defaults to all records.
  HealthStats getStats({int? lastNDays}) {
    final records = lastNDays == null
        ? getAllRecords()
        : getRecordsLastNDays(lastNDays);

    final weights = records
        .map((r) => r.weight)
        .whereType<double>()
        .toList();
    final systolics = records
        .map((r) => r.systolic)
        .whereType<int>()
        .toList();
    final diastolics = records
        .map((r) => r.diastolic)
        .whereType<int>()
        .toList();
    final heartRates = records
        .map((r) => r.heartRate)
        .whereType<int>()
        .toList();
    final stepsList =
        records.map((r) => r.steps).whereType<int>().toList();
    final sleepHoursList = records
        .map((r) => r.sleepHours)
        .whereType<double>()
        .toList();

    return HealthStats(
      recordCount: records.length,
      weightCount: weights.length,
      avgWeight: _avgDouble(weights),
      minWeight: weights.isEmpty
          ? null
          : weights.reduce((a, b) => a < b ? a : b),
      maxWeight: weights.isEmpty
          ? null
          : weights.reduce((a, b) => a > b ? a : b),
      bloodPressureCount: systolics.length,
      avgSystolic: _avgInt(systolics),
      avgDiastolic: _avgInt(diastolics),
      heartRateCount: heartRates.length,
      avgHeartRate: _avgInt(heartRates),
      stepsCount: stepsList.length,
      totalSteps: stepsList.fold<int>(0, (a, b) => a + b),
      sleepCount: sleepHoursList.length,
      avgSleepHours: _avgDouble(sleepHoursList),
    );
  }

  static double? _avgDouble(List<double> xs) {
    if (xs.isEmpty) return null;
    return xs.reduce((a, b) => a + b) / xs.length;
  }

  static double? _avgInt(List<int> xs) {
    if (xs.isEmpty) return null;
    return xs.reduce((a, b) => a + b) / xs.length;
  }

  // ----------------------------------------------------------------- writes

  /// Inserts or updates. Generates id when input id is empty.
  Future<HealthRecord> saveRecord(HealthRecord record) async {
    final withId =
        record.id.isEmpty ? record.copyWith(id: _generateId()) : record;

    await _store.clear(withId.id, collectionName);

    final chunk = DocChunk(
      docName: withId.id,
      chunkIndex: 0,
      text: withId.encode(),
      embedding: const [],
      collectionName: collectionName,
    );
    await _store.addToCollection(collectionName, chunk);
    notifyListeners();
    return withId;
  }

  Future<bool> deleteRecord(String id) async {
    final prevCount = count;
    await _store.clear(id, collectionName);
    if (count < prevCount) {
      notifyListeners();
      return true;
    }
    return false;
  }

  Future<int> deleteAllRecords() async {
    final prevCount = count;
    await _store.deleteCollection(collectionName);
    if (prevCount > 0) notifyListeners();
    return prevCount;
  }

  // -------------------------------------------------------------- internals

  String _generateId() {
    final ts = DateTime.now().microsecondsSinceEpoch;
    final rand = _rng.nextInt(0xFFFFFFFF).toRadixString(16).padLeft(8, '0');
    return '$_idPrefix${ts}_$rand';
  }
}

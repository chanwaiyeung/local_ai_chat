import 'package:flutter_test/flutter_test.dart';
import 'package:local_ai_chat/services/rrf_tuner.dart';

void main() {
  test('generateGrid creates normalized RRF configs', () {
    final configs = const RrfTuner().generateGrid();

    expect(configs, isNotEmpty);
    expect(configs.length, 36);
    for (final config in configs) {
      expect(
        config.semanticWeight + config.keywordWeight,
        closeTo(1.0, 0.01),
      );
    }
  });

  test('rrfScore decreases as rank increases', () {
    final tuner = const RrfTuner();

    expect(tuner.rrfScore(1, 60), greaterThan(tuner.rrfScore(2, 60)));
  });

  test('bestResult prefers pass rate then score', () {
    final tuner = const RrfTuner();
    final configs = tuner.generateGrid();

    final best = tuner.bestResult([
      RrfTuningResult(
        config: configs[0],
        passRate: 0.9,
        score: 10,
        pass: 9,
        partial: 2,
        fail: 1,
      ),
      RrfTuningResult(
        config: configs[1],
        passRate: 0.95,
        score: 9,
        pass: 10,
        partial: 0,
        fail: 1,
      ),
    ]);

    expect(best.config, configs[1]);
  });
}

import 'package:flutter_test/flutter_test.dart';
import 'package:local_ai_chat/controllers/wealth_controller.dart';
import 'package:local_ai_chat/models/wealth_record.dart';
import 'package:local_ai_chat/services/vector_store.dart';

void main() {
  late VectorStore store;
  late WealthController controller;

  setUp(() async {
    store = VectorStore();
    controller = WealthController(store);
    // wait for initial load if needed, though VectorStore is sync for now
  });

  test('can save and retrieve wealth records', () async {
    final record = WealthRecord(
      date: DateTime(2026, 5, 1),
      assetType: 'stock',
      amount: 10000,
      currency: 'USD',
      notes: 'Apple',
    );

    await controller.saveRecord(record);
    final records = controller.getAllRecords();

    expect(records.length, 1);
    expect(records.first.assetType, 'stock');
    expect(records.first.amount, 10000);
  });

  test('can search records', () async {
    await controller.saveRecord(WealthRecord(
      date: DateTime.now(),
      assetType: 'fund',
      amount: 5000,
      notes: 'Vanguard',
    ));

    final results = controller.searchRecords('vanguard');
    expect(results.length, 1);
    expect(results.first.assetType, 'fund');
  });

  test('calculates total assets', () async {
    await controller.saveRecord(WealthRecord(
      date: DateTime.now(),
      assetType: 'cash',
      amount: 50000,
    ));
    await controller.saveRecord(WealthRecord(
      date: DateTime.now(),
      assetType: 'stock',
      amount: 150000,
    ));

    final total = controller.getTotalAssets();
    expect(total, 200000);
  });

  test('calculates asset allocation', () async {
    await controller.saveRecord(WealthRecord(
      date: DateTime.now(),
      assetType: 'cash',
      amount: 50000,
    ));
    await controller.saveRecord(WealthRecord(
      date: DateTime.now(),
      assetType: 'stock',
      amount: 150000,
    ));

    final allocation = controller.getAssetAllocation();
    expect(allocation['cash'], 50000);
    expect(allocation['stock'], 150000);
  });
}
